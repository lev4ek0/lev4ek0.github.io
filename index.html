                import numpy as np

                def decorate_matrix(n):
                    A = np.ones((n, n))
                    B = np.zeros((n-2, n-2))
                    A[1:n-1, 1:n-1] = B
                    return 'array({})'.format(',\n'.join(map(str, A))).replace(' ', ', ')

                n = input()

                print(decorate_matrix(int(n)))
